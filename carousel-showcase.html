Full-Width Image Slider

HTML + CSS + JS 
<!-- HTML Structure -->
<div class="carousel">
  <div class="track" id="track">
    <div class="slide"><img src="photo1.jpg" alt="Slide 1"></div>
    <div class="slide"><img src="photo2.jpg" alt="Slide 2"></div>
  </div>
  <button id="prev">←</button>
  <button id="next">→</button>
</div>

/* CSS */
.carousel { overflow: hidden; }
.track    { display: flex; transition: transform 0.6s cubic-bezier(.77,0,.175,1); }
.slide    { min-width: 100%; }

// JavaScript
let idx = 0;
const track = document.getElementById('track');
const total = document.querySelectorAll('.slide').length;

function goTo(n) {
  idx = ((n % total) + total) % total; // wrap around
  track.style.transform = `translateX(-${idx * 100}%)`;
}
document.getElementById('next').onclick = () => goTo(idx + 1);
document.getElementById('prev').onclick = () => goTo(idx - 1);

PropertyValueNotes
AnimationCSS opacityAll slides stacked; only 1 visible
Best forTestimonials, moody visualsNo spatial confusion
Autoplay✓ YesWorks great with timer bar
Slide direction✗ NonePure opacity — no movement
DifficultyEasySimplest JS logic of all types
CSS KEY RULE + JS 
/* All slides stacked on top of each other */
.slide        { position: absolute; inset: 0; opacity: 0; transition: opacity 0.8s ease; }
.slide.active { opacity: 1; }   /* toggle this class */

// JavaScript — just swap the .active class
let current = 0;
const slides = document.querySelectorAll('.slide');

function fadeToSlide(n) {
  slides[current].classList.remove('active');
  current = (n + slides.length) % slides.length;
  slides[current].classList.add('active');
}

// Auto-advance every 4 seconds
setInterval(() => fadeToSlide(current + 1), 4000);

MULTI-CARD KEY CSS 
/* Show 3 cards; peek at edges with padding */
.viewport { overflow: hidden; }
.track    { display: flex; gap: 1.2rem; transition: transform 0.5s ease; }

/* Each card takes up exactly 1/3 of viewport */
.card     { min-width: calc(33.333% - 0.8rem); }

// JavaScript — move by one card width
let pos = 0;
const cardW = track.children[0].offsetWidth + 19; // + gap
const maxPos = (cards.length - 3) * cardW;

nextBtn.onclick = () => {
  pos = Math.min(pos + cardW, maxPos);
  track.style.transform = `translateX(-${pos}px)`;
};

THUMBNAIL SYNC LOGIC 
// Sync thumbnails with main carousel
const thumbs = document.querySelectorAll('.thumb');

function goToSlide(n) {
  // Move main track
  mainTrack.style.transform = `translateX(-${n * 100}%)`;

  // Update active thumbnail
  thumbs.forEach((t, i) => {
    t.classList.toggle('active', i === n);
  });
  current = n;
}

// Click any thumbnail → jump to that slide
thumbs.forEach((thumb, i) => {
  thumb.onclick = () => goToSlide(i);
});

KEN BURNS EFFECT 
/* The Ken Burns zoom: start zoomed in, slowly zoom to normal */
.slide .bg        { transform: scale(1.08); transition: transform 8s ease-out; }
.slide.active .bg { transform: scale(1); }  /* zoom OUT slowly */

/* Crossfade between slides */
.slide        { position: absolute; inset: 0; opacity: 0; transition: opacity 1s ease; }
.slide.active { opacity: 1; }

// Progress bar drives the autoplay
function startProgress() {
  bar.style.transition = 'none';
  bar.style.width = '0%';
  setTimeout(() => {
    bar.style.transition = 'width 6s linear';
    bar.style.width = '100%';
  }, 50);
}
setInterval(() => { nextSlide(); startProgress(); }, 6000);
